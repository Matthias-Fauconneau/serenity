43 inline bool intersects(const float radius, const float3 origin, const float3 ray, float& tmin, float& tmax) {
44  float a = sq(ray);
45  float b = 2 * dot(ray, origin);
46  float c = sq(origin) - sq(radius);
47  float d2 = sq(b) - 4 * a * c;
48  if(d2 < 0) return false;
49  float d = sqrt(d2);
50  float q = b < 0 ? (-b - d)/2 : (-b + d)/2;
51  tmin = q / a, tmax = c / q;
52  if(tmin > tmax) swap(tmin, tmax);
