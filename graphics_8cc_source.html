103 void line(const Image& target, float x1, float y1, float x2, float y2, vec3 color, float alpha) {
104  color = clip(vec3(0.f), color, vec3(1));
105  float dx = x2 - x1, dy = y2 - y1;
106  bool transpose=false;
107  if(abs(dx) < abs(dy)) { swap(x1, y1); swap(x2, y2); swap(dx, dy); transpose=true; }
108  if(x2 < x1) { swap(x1, x2); swap(y1, y2); }
109  if(dx==0) return; //p1==p2
110  float gradient = dy / dx;
111  int i1,i2; float intery;
112  {
113  float xend = round(x1), yend = y1 + gradient * (xend - x1);
114  float xgap = 1 - fract(x1 + 1./2);
115  blend(target, xend, yend, color, (1-fract(yend)) * xgap * alpha, transpose);
116  blend(target, xend, yend+1, color, fract(yend) * xgap * alpha, transpose);
117  i1 = int(xend);
118  intery = yend + gradient;
119  }
120  {
121  float xend = round(x2), yend = y2 + gradient * (xend - x2);
122  float xgap = fract(x2 + 1./2);
123  blend(target, xend, yend, color, (1-fract(yend)) * xgap * alpha, transpose);
124  blend(target, xend, yend+1, color, fract(yend) * xgap * alpha, transpose);
125  i2 = int(xend);
126  }
127  for(int x=i1+1;x<i2;x++) {
128  blend(target, x, intery, color, (1-fract(intery)) * alpha, transpose);
129  blend(target, x, intery+1, color, fract(intery) * alpha, transpose);

131  }
